224  ConstAdminIterator(const AdminPairingInfo * start, size_t index, size_t maxSize) :
225  mStart(start), mIndex(index), mMaxSize(maxSize)
226  {
227  if (mIndex >= maxSize)
228  {
229  mIndex = maxSize;
230  }
231  else if (!mStart[mIndex].IsInitialized())
232  {
233  Advance();
234  }
235  }
236  ConstAdminIterator(const ConstAdminIterator &) = default;
237  ConstAdminIterator & operator=(const ConstAdminIterator &) = default;
238 
239  ConstAdminIterator & operator++() { return Advance(); }
240  ConstAdminIterator operator++(int)
241  {
242  ConstAdminIterator other(*this);
243  Advance();
244  return other;
245  }
246 
247  const AdminPairingInfo & operator*() const { return mStart[mIndex]; }
248  const AdminPairingInfo * operator->() const { return mStart + mIndex; }
249 
250  bool operator==(const ConstAdminIterator & other)
251  {
252  if (IsAtEnd())
253  {
254  return other.IsAtEnd();
255  }
256 
257  return (mStart == other.mStart) && (mIndex == other.mIndex) && (mMaxSize == other.mMaxSize);
258  }
259  bool operator!=(const ConstAdminIterator & other) { return !(*this == other); }
260 
261  bool IsAtEnd() const { return (mIndex == mMaxSize); }
262 
263 private:
264  const AdminPairingInfo * mStart;
265  size_t mIndex;
266  size_t mMaxSize;
267 
268  ConstAdminIterator & Advance()
269  {
270  do
271  {
272  if (mIndex < mMaxSize)
273  {
274  mIndex++;
275  }
276  } while (!IsAtEnd() && !mStart[mIndex].IsInitialized());
277 
278  return *this;
279  }
