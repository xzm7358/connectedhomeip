27 template <typename T>
28 bool IsCurrentInterfaceUsable(T & iterator)
29 {
30  if (!iterator.IsUp() || !iterator.SupportsMulticast())
31  {
32  return false; // not a usable interface
33  }
34  char name[chip::Inet::InterfaceIterator::kMaxIfNameLength];
35  if (iterator.GetInterfaceName(name, sizeof(name)) != CHIP_NO_ERROR)
36  {
37  ChipLogError(Discovery, "Failed to get interface name.");
38  return false;
39  }
40 
41  // TODO: need a better way to ignore local loopback interfaces/addresses
42  // We do not want to listen on local loopback even though they are up and
43  // support multicast
44  //
45  // Some way to detect 'is local looback' that is smarter (e.g. at least
46  // strict string compare on linux instead of substring) would be better.
47  //
48  // This would reject likely valid interfaces like 'lollipop' or 'lostinspace'
49  if (strncmp(name, "lo", 2) == 0)
50  {
52  return false;
53  }
54  return true;
55 }

57 class AllInterfaces : public mdns::Minimal::ListenIterator
58 {
59 private:
60 public:
61  AllInterfaces() { SkipToFirstValidInterface(); }

63  bool Next(chip::Inet::InterfaceId * id, chip::Inet::IPAddressType * type) override
64  {
65  if (!mIterator.HasCurrent())
66  {
67  return false;
68  }
69 
70 #if INET_CONFIG_ENABLE_IPV4
71  if (mState == State::kIpV4)
72  {
73  *id = mIterator.GetInterfaceId();
74  *type = chip::Inet::kIPAddressType_IPv4;
75  mState = State::kIpV6;
76  return true;
77  }
78 #endif
79 
80  *id = mIterator.GetInterfaceId();
81  *type = chip::Inet::kIPAddressType_IPv6;
82 #if INET_CONFIG_ENABLE_IPV4
83  mState = State::kIpV4;
84 #endif
85 
86  for (mIterator.Next(); SkipCurrentInterface(); mIterator.Next())
87  {
88  }
89  return true;
90  }
91 
92 private:
93  enum class State
94  {
95  kIpV4,
96  kIpV6,
97  };
98 #if INET_CONFIG_ENABLE_IPV4
99  State mState = State::kIpV4;
100 #else
101  State mState = State::kIpV6;
102 #endif
103  chip::Inet::InterfaceIterator mIterator;
104 
105  void SkipToFirstValidInterface()
106  {
107  do
108  {
109  if (!SkipCurrentInterface())
110  {
111  break;
112  }
113  } while (mIterator.Next());
114  }
115 
116  bool SkipCurrentInterface()
117  {
118  if (!mIterator.HasCurrent())
119  {
120  return false; // nothing to try.
121  }
122 
123  return !IsCurrentInterfaceUsable(mIterator);
124  }
125 };
