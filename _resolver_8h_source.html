44 
45  char hostName[kHostNameSize + 1];

47  uint16_t vendorId;
48  uint16_t productId;
49  int numIPs;
50  Inet::IPAddress ipAddress[kMaxIPAddresses];
51  void Reset()
52  {
53  memset(hostName, 0, sizeof(hostName));
54  longDiscriminator = 0;
55  vendorId = 0;
56  productId = 0;
57  numIPs = 0;
58  for (int i = 0; i < kMaxIPAddresses; ++i)
59  {
60  ipAddress[i] = chip::Inet::IPAddress::Any;
61  }
62  }
63  CommissionableNodeData() { Reset(); }
64  bool IsHost(const char * host) const { return strcmp(host, hostName) == 0; }
65  bool IsValid() const { return !IsHost("") && ipAddress[0] != chip::Inet::IPAddress::Any; }
